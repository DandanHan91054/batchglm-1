This file gives an overview of the classes and inheritance relations in batchglm. Classes are listed by directory and file. Refer to the documenation on details on the classes. Used name_scopes are listed here.

- models
    - base.py
        - BasicInputData
        - BasicModel
        - BasicEstimator <- .BasicModel
        - BasicSimulator <- .BasicModel

    - nb
        - base.py
            - InputData <- .base.BasicInputData
            - Model <- .base.BasicModel
            - XArrayModel <- .Model
            - AbstractEstimator <- .Model
            - XArrayEstimatorStore <- .AbstractEstimator
        - simulator.py:
            - Simulator <- .base.BasicSimulator

    - glm
        - base.py
            - Model <- BasicModel
            
    - nbglm
        - base.py
            - InputData <- .nb.InputData
            - Model <- .nb.Model, .glm.Model
            - XArrayModel <- .Model
            - AbstractEstimator <- .Model
            - XArrayEstimatorStore <- .nb.AbstractEstimator, .nb.XArrayModel
            - simulator.py:
            - Simulator <- .nb.Simulator
